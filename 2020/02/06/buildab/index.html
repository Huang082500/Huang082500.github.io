Unity UI Text 外文显示单词适配

2020-11-01 次阅读

  • 问题
  • 解决思路
  • 代码实现

问题

Unity UI Text 组件显示外文内容时会发生单词被截断

解决思路

当前句子加上最后一个单词的总宽度大于 Text 控件的宽度时，就将该句的最后一个单词放到下一句显示。

        //获取设置过换行符的文字
        //str:原文字ss
        //width:Text 空间的宽度
        //fontSize: 字体大小
        //text:当前的 Text 引用 sss
        public static string GetText(string str, float width,int fontSize,Text text)
        {
            char[] chars = str.ToCharArray();
            string tempStr = "";
            StringBuilder stringBuilder = new StringBuilder();
            StringBuilder tempStringBuilder = new StringBuilder();
            float tempStrLength = 0;
            text.font.RequestCharactersInTexture(str, fontSize, FontStyle.Normal);  //指定需要渲染哪些字符
            for (int i = 0; i < chars.Length; i++)
            {
                if (chars[i] == '\r' || chars[i] == '\n')
                {
                    tempStrLength = 0;
                    if (tempStringBuilder.Length>0)
                    {
                        stringBuilder.Append(tempStringBuilder);
                        tempStringBuilder.Remove(0, tempStringBuilder.Length);
                    }
                    stringBuilder.Append(chars[i]);
                }
                else
                {
                    if (tempStrLength >= width)
                    {
                        stringBuilder.Append('\n');
                        tempStrLength = 0;
                    }
                    if (chars[i] == ' ')
                    {
                        stringBuilder.Append(tempStringBuilder);
                        stringBuilder.Append(chars[i]);
                        tempStringBuilder.Remove(0, tempStringBuilder.Length);
                    }
                    else
                    {
                        tempStringBuilder.Append(chars[i]);
                    }
                }
                float length = GetTextContentWidth(chars[i], text, fontSize);
                //Debug.Log("char:   |"+chars[i]+"|  .length:"+length);
                tempStrLength += length;
            }
            return stringBuilder.ToString();
        }


        //获取文字的宽度
        public static float GetTextContentWidth(char ch, Text text, int fontSize)
        {
            CharacterInfo info = new CharacterInfo();
            float textSize = 0f;
            if (text)
            {
                bool isHave = text.font.GetCharacterInfo(ch, out info, fontSize);  //获取每个字符的渲染信息
                textSize = info.advance;                                           //当前字符原点到下一字符原点的距离（以像素为单位）
               
            }
            return textSize;
        }

坑点

原本以为可以直接通过 GetCharacterInfo() 方法获取当前字符宽度信息，这样直接调用的结果是返回的 CharacterInfo 为空，后来发现需要先去调用 RequestCharactersInTexture() 方法来指定需要渲染哪些字符，然后才能通过 GetCharacterInfo() 方法获取当前字符的宽度信息。
