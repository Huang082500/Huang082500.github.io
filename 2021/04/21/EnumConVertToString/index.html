using System;
public enum Test
{
      Aaa = 1,
      Bbb = 2,
}
public class func
{
    public void TestFunciton()
    {
        var configStr = "abc";
        if (Enum.IsDefined(typeof(Test), configStr))  //判断当前字符串是否在枚举中存在
        {
            var Id = (Test)Enum.Parse(typeof(Test), configStr);  //取出当前字符串在枚举中对应的值
        }
    }
    public void TestFuncitonTwo()
    {
        string configStr = Enum.GetName(typeof(Test), Test.Aaa);   //枚举转字符串
    }
}
